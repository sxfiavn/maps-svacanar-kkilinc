1   package edu.brown.cs.student.main.csv.creators;

3   import edu.brown.cs.student.main.csv.FactoryFailureException;
4   import java.util.List;
5   
6   /**
7    * The DataTypeB class of this project. This is a STARS class that implements CreatorFromRow. It is
8    * suitable for a CSV that contains StarID, Proper Name, X, Y, Z as column names
9    */
10  public class CreateStar implements CreatorFromRow<CreateStar> {
11    private List<String> row;
12    private int starId;
13    private String properName;
14    private double x1;
15    private double y1;
16    private double z1;
17  
18    /** The DataTypeB default constructor takes no arguments. */
19    public CreateStar() {}
20  
21    /**
22     * The DataTypeB constructor takes in a row.
23     *
24     * @param row - a list of strings
25     */
26    public CreateStar(List<String> row) throws FactoryFailureException {
27      if (row.size() != 5) {
28        throw new FactoryFailureException("Incorrect number of columns", row);
29      }
30      this.row = row;
31      this.starId = Integer.parseInt(row.get(0));
32      this.properName = row.get(1);
33      this.x1 = Double.parseDouble(row.get(2));
34      this.y1 = Double.parseDouble(row.get(3));
35      this.z1 = Double.parseDouble(row.get(4));
36    }
37  
38    /**
39     * The create method.
40     *
41     * @param row - a list of strings
42     * @return - an object of type DataTypeA
43     * @throws FactoryFailureException - when an empty row is given
44     */
45    @Override
46    public CreateStar create(List<String> row) throws FactoryFailureException {
47      return new CreateStar(row);
48    }
49  
50    /**
51     * Changes the row into a string object with its name in front.
52     *
53     * @return - a string of the rows contents.
54     */
55    public String toString() {
56      return "StarID: "
57          + this.starId
58          + ", Proper Name: "
59          + this.properName
60          + ", X: "
61          + this.x1
62          + ", Y: "
63          + this.y1
64          + ", Z: "
65          + this.z1;
66    }
67  
68    /**
69     * The get method.
70     *
71     * @param index - an integer representing the row of interest
72     * @return - String of the object at that location
73     */
74    public String get(int index) {
75      return this.row.get(index);
76    }
77  
78    /**
79     * The contains method checks if a string is in the row.
80     *
81     * @param str - a string representing the target.
82     * @return - a boolean of whether the string is in the row.
83     */
84    public boolean contains(String str) {
85      return this.row.contains(str);
86    }
87  
88    /**
89     * The size method returns the size of the row.
90     *
91     * @return - an integer of the size of the row.
92     */
93    public int size() {
94      return this.row.size();
95    }
96  }
