76     * Method that returns a nested list that can later be iterated through in order to get the code

78     *
79     * @return A list of lists representing correspondence between state names and their codes.
80     * @throws DatasourceException that is triggered if the API server throws an exception.
81     */
82    public static List<List<String>> apiStateNamesToCodes() throws DatasourceException {
83      try {
84        System.out.println("Connecting to ACS API for StateNamesToCodes Data...");
85        URL requestURL =
86            new URL("https", "api.census.gov", "/data/2010/dec/sf1?get=NAME&for=state:*");
87        Type listOfLists = Types.newParameterizedType(List.class, List.class);
88        return MoshiUtils.fetchAndDeserialize(requestURL, listOfLists);
89      } catch (IOException e) {
90        System.out.println(e.getMessage());
91        throw new DatasourceException("Exception occurred");
92      }
93    }
