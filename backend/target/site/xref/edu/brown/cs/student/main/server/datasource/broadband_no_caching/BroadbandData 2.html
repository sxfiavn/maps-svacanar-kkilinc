5   /**
6    * Broadband record holds details of the percentage of coverage, state, county information, and
7    * possible errors in obtaining the data.
8    */
9   public record BroadbandData(
10      String name,
11      String percentage,
12      String state,
13      String county,
14      String errorMessage,
15      String retrievalTime) {
16  
17    /**
18     * Factory method of BroadbandData for error scenarios.
19     *
20     * @param errorMessage A string with the error message.
21     * @return A BroadbandData instance with the error message and null for all other fields.
22     */
23    public static BroadbandData forErrorMessage(String errorMessage, String time) {
24      return new BroadbandData(null, null, null, null, errorMessage, time);
25    }
26  
27    /**
28     * Factory method of BroadbandData that separates the given List into its respective name,
29     * percentage, state, and county data.
30     *
31     * @param broadbandDataValue A nested list containing broadband data.
32     * @return A BroadbandData instance with the given data.
33     */
34    public static BroadbandData fromData(List<List<String>> broadbandDataValue, String time) {
35      if (broadbandDataValue != null && broadbandDataValue.size() >= 2) {
36        List<String> values = broadbandDataValue.get(1);
37        return new BroadbandData(
38            values.get(0), values.get(1), values.get(2), values.get(3), null, time);
39      }
40      return BroadbandData.forErrorMessage("No data returned from API", time);
41    }
42  }
