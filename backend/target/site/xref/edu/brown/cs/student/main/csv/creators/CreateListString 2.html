5   /** This class converts each row of the CSV into a string. */
6   public class CreateListString implements CreatorFromRow<String> {
7   
8     public CreateListString() {}
9   
10    /**
11     * This is a method that will take in a row from the Parser in list form, then iterate over it,
12     * concatenate the elements, and return the combined String output in one continuous line.
13     *
14     * @param row row passed in from the Parser
15     * @return a String that represents just the concatenation of the elements of the passed in list
16     */
17    @Override
18    public String create(List<String> row) {
19      // first wrap the inputted list -> array list of strings for flexible access to elements
20      StringBuilder out =
21          new StringBuilder(); // StringBuilder is much more memory efficient than just concatenating
22      // strings
23      for (int i = 0; i < row.size(); i++) {
24        if (i == 0) { // each new row (doesn't have extra comma)
25          out.append(row.get(i));
26        } else {
27          out.append(", ").append(row.get(i)); // just append each string to the final output string
28        }
29      }
30      return out
31          .toString(); // no try catch here because i'll just assume there's a string input no matter
32    }
33  }
