3   import edu.brown.cs.student.main.csv.FactoryFailureException;
4   import java.util.ArrayList;
5   import java.util.List;

7   /** This class converts each row in the CSV into a list of booleans. */
8   public class CreateBoolean implements CreatorFromRow<List<Boolean>> {
9   
10    public CreateBoolean() {}
11  
12    /**
13     * This converts a passed in row to a list of booleans, using case-insensitive matching for
14     * booleans (Java's naming convention) to constrain the otherwise lax criteria set by
15     * .parseBoolean(string)
16     *
17     * @param row passed in row parsed from the CSV file, from the Parser class
18     * @return a new list object that converts the passed in row to a list of booleans
19     * @throws FactoryFailureException an exception thrown if any arbitrary string in the passed in
20     *     row is not parseable as a boolean per the conditions set
21     */
22    @Override
23    public List<Boolean> create(List<String> row) throws FactoryFailureException {
24      // just wrap the inputted list -> array list of strings for flexible access
25      try {
26        ArrayList<Boolean> bools = new ArrayList<>();
27        for (String s : row) {
28          if (!s.isEmpty() && (s.equals("true") || s.equals("false"))) {
29            Boolean b = Boolean.parseBoolean(s); // convert the string into a boolean
30            bools.add(b); // add to the list of booleans
31          } else {
32            throw new IllegalArgumentException(
33                "Reformat boolean string to be: true or false, for the following input string. "
34                    + "If nothing is shown, the string passed was empty: "
35                    + s);
36          }
37        }
38        return bools;
39      } catch (IllegalArgumentException i) {
40        throw new FactoryFailureException(
41            "Inputted row has a string that is in a invalid boolean format: " + i.getMessage(), row);
42      }
43    }
44  }
