3   import edu.brown.cs.student.main.csv.FactoryFailureException;
4   import java.util.ArrayList;
5   import java.util.List;
6   
7   /** This class converts each row in the CSV into a list of integers. */
8   public class CreateListInt implements CreatorFromRow<List<Integer>> {
9   
10    public CreateListInt() {}
11  
12    /**
13     * This method takes in a row from the Parser in list form, iterates over it, converts each string
14     * to its integer counterpart if possible, then appends the elements to a new list, which is then
15     * returned.
16     *
17     * @param row row passed in from the Parser in list form
18     * @return a list of integers that the original row was converted to
19     * @throws FactoryFailureException if any arbitrary string in the passed in row cannot be
20     *     parsed/converted into an integer
21     */
22    @Override
23    public List<Integer> create(List<String> row) throws FactoryFailureException {
24  
25      // just wrap the inputted list -> array list of strings for flexible access
26      ArrayList<Integer> ints = new ArrayList<>();
27      try {
28        for (String s : row) {
29          Integer i = Integer.parseInt(s);
30          ints.add(i);
31        }
32      } catch (NumberFormatException nf) {
33        throw new FactoryFailureException(
34            "Inputted row has strings that can't be parsed into an integer: " + nf.getMessage(), row);
35      }
36      return ints;
37    }
38  }
